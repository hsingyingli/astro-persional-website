# Description
***Medium***
```
Given a string s, return the longest palindromic substring in s.

A string is called a palindrome string if the reverse of that string is the same as the original string.
```

# TypeScript
```typescript
function longestPalindrome(s: string): string {
    let maxSubString = ""
    
    for (let i: number = 0; i<s.length; i++) {
        let sub1 = isPalindromicSubString(s, i, i+1)
        let sub2 = isPalindromicSubString(s, i, i)

        if (sub1.length > maxSubString.length) {
            maxSubString = sub1
        }
        if (sub2.length > maxSubString.length) {
            maxSubString = sub2
        }
    }

    return maxSubString
};

function isPalindromicSubString(s: string, left: number, right: number): string {
    let subString = ""
    while (left >= 0 && right < s.length && s[left] === s[right]) {
        subString = s.substring(left, right+1)
        left--
        right++
    }
    
    return subString
}
```

# Python3
```python
class Solution:
    def longestPalindrome(self, s: str) -> str:
        self.s = s
        res = ''
        
        for i in range(len(s)):
            sub1 = self.check(i, i)
            sub2 = self.check(i, i+1)
            res = sub1 if len(sub1) > len(res) else res 
            res = sub2 if len(sub2) > len(res) else res 
        return res
        
    def check(self, l, r):
        subs = ''
        while l>=0 and r<len(self.s) and self.s[l] == self.s[r]:
            if r - l + 1 > len(subs) :
                subs = self.s[l:r+1]
            l-=1
            r+=1
        return subs
```
