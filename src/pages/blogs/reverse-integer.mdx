## Description
***Medium***
```
Given a signed 32-bit integer x, return x with its digits reversed. If reversing x causes the value to go outside the signed 32-bit integer range [-231, 231 - 1], then return 0.

Assume the environment does not allow you to store 64-bit integers (signed or unsigned).
```

## TypeScript
```typescript
function reverse(x: number): number {
    if (x === 0 ) return x 
    const sign: number = x > 0 ? 1 : -1 
    x = Math.abs(x)
    let ans: number = x % 10 
    x = Math.floor(x / 10) 
    
    while(x) {
        ans *= 10
        ans += x % 10
        x = Math.floor(x / 10) 
    }
    if (ans > Math.pow(2, 31)) return 0
    return ans * sign
};
```

## Python3
```python
class Solution:
    def reverse(self, x: int) -> int:
        sign = 1 if x > 0 else -1
        ans = int(str(abs(x))[::-1])
        if (ans > pow(2, 31)-1): return 0
        return ans * sign
```

## Golang
```go
func reverse(x int) int {
    if x == 0 {
        return 0
    }

    sign, num := abs(x)
    ans := num %10
    num = num / 10

    for num > 0 {
        ans *= 10
        ans += num % 10 
        num = num / 10
    }

    if ans > math.MaxInt32 {
        return 0
    }

    return ans * sign
}


func abs(x int) (sign int, num int ){
    if x > 0 {
        sign = 1
        num = x
    } else {
        sign = -1
        num = -x
    }
    return sign, num
}
```
